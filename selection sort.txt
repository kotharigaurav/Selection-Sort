void sort(int a[],int size)
{
 int i,j,min,index;
 for(i=0;i<size;i++)
 {
  min=a[i];
  index=i;
  for(j=i+1;j<size;j++)
  {
   if(min>a[j])
   {
   min=a[j];
   index=j;
  }
 }
 a[index]=a[i];
 a[i]=min;
 }
}
